<script src="./oglib.js"></script>
<script type="text/javascript" src="https://cdnjs.cloudflare.com/ajax/libs/psl/1.8.0/psl.min.js"></script>
<link rel="stylesheet" href="og.css">

ogfind.ogNewMsg(msg,display);
